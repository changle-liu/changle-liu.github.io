def CrossEntropy_loss(y_true:list,y_pred:list):
    """
    y_true,y_predÔºåÂàÜÂà´ÊòØ‰∏§‰∏™Ê¶ÇÁéáÂàÜÂ∏É

    assert len(y_true)==len(y_pred)
    loss=0
    for y,fx in zip(y_true,y_pred):
        loss+=-y * np.log(fx)
    return loss
